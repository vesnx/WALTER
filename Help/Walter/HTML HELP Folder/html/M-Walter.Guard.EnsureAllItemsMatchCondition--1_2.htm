This test method demonstrates the validation logic catching an invalid order and throwing an ArgumentMatchConditionException, which is then expected and handled within the test framework.
[TestClass()]
   public class GuardTests
   {
       [TestMethod()]
       [ExpectedException(typeof(ArgumentMatchConditionException))]
       public void EnsureAllItemsMatchConditionTest()
       {
           Order[] myOrders = new Order[] {
               new Order(100, "SKU1"),
               new Order(200, "SKU2"),
               new Order(-30, "SKU1"), // This will cause the exception
               new Order(400, "SK4")
           };
           int clientId = 123;

           // Attempt to process orders, expecting an exception due to the negative quantity
           ProcessOrders(myOrders.AsSpan(), clientId: clientId);
       }

       public void ProcessOrders(Span&lt;Order&gt; ordersToCharge, int clientId)
       {
           // Only allow items that have a positive quantity; else, we would refund/negative charge
           Guard.EnsureAllItemsMatchCondition(ordersToCharge, item => item.Quantity &gt; 0);
       }
   }
