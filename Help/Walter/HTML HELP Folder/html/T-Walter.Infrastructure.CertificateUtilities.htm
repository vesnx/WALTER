Remarks

Being able to test if a file is signed without the use of a X509Certificate2 is beneficial as it reduces the number or exceptions generated by your application if a non-signed file is opened

Examples

Console Application Example: Check File Signature Interactively
This console application prompts the user to enter a file path and then checks if the file is digitally signed using the IsFileSigned method from the CertificateUtilities class. It demonstrates a practical use case where someone could verify the signature status of arbitrary files without hardcoding the paths.
This example provides a simple yet effective demonstration of how to use the IsFileSigned method within an interactive console application, offering a practical tool for end-users or administrators needing to verify file signatures on-demand
using System;
   using System.Runtime.InteropServices;

   namespace SignatureVerificationUtility
   {
       using Walter.Infrastructure;
       class Program
       {
           static void Main(string[] args)
           {
               Console.WriteLine("Signature Verification Utility");
               Console.WriteLine("Enter the path of the file you want to check, or type 'exit' to close:");

               while (true)
               {
                   Console.Write("\nFile path: ");
                   var filePath = Console.ReadLine();

                   // Exit condition
                   if (string.Equals(filePath, "exit", StringComparison.OrdinalIgnoreCase))
                   {
                       break;
                   }

                   if (!string.IsNullOrEmpty(filePath))
                   {
                       // Check if running on Windows
                       if (RuntimeInformation.IsOSPlatform(OSPlatform.Windows))
                       {
                           try
                           {
                               bool isSigned = CertificateUtilities.IsFileSigned(filePath);
                               Console.WriteLine(isSigned ? "The file is digitally signed." : "The file is not digitally signed or the signature could not be verified.");
                           }
                           catch (Exception ex)
                           {
                               Console.WriteLine($"An error occurred: {ex.Message}");
                           }
                       }
                       else
                       {
                           Console.WriteLine("This utility runs only on Windows due to specific API dependencies.");
                           break;
                       }
                   }
                   else
                   {
                       Console.WriteLine("Invalid input. Please enter a valid file path.");
                   }
               }
           }
       }
   }
